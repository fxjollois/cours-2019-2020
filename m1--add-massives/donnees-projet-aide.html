import pymongo
import pandas

con = pymongo.MongoClient("mongodb://193.51.82.104:2343/")
horo = con.horodateurs
mob = horo.mobiliers

res = mob.aggregate([
    { "$group" : { "_id" : "$fields.arrondt", "nb" : { "$sum" : 1}}},
    { "$sort" : { "_id": 1}}
])
df = pandas.DataFrame(list(res)) 
df.columns = ["Arrondissement", "Effectif"]
df.head()
